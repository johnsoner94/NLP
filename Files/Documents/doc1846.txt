        function Validate() {
            $.ajax({
                url: '/Admin/login/islockeduser',
                dataType: 'json',
                data: {
                    email: $('#loginForm #Login').val()
                },
                success: function (data) {
                    if (data.isLockedUser) {
                        $('#ServerErrors span').text(data.error);
                        $('#ServerErrors span').show();
                    } else {
                        tryLogin();
                    }
                },
                async: false

        function tryLogin(parameters) {
            var form = $('#loginForm');
            if (form.valid()) {
                $.post("/Account/LoginCandidateByEmail", form.serialize(), function (response) {
                    if (!response.isSuccess &amp;&amp; !response.isFullPending &amp;&amp; !response.isDeny) {
                        $('#ServerErrors span').text(response.error);
                        $('#ServerErrors span').show();
                    } else if (!response.isSuccess &amp;&amp; response.isFullPending) {
                        loadPopup('pendingDenyPopup');
                    } else if (!response.isSuccess &amp;&amp; response.isDeny) {
                        loadPopup('denyUserPopup');
                    }
                    else {
                        window.location.assign(response.redirectUrl);
                    }
                });
            } else {
                $('#clienterrors').show();
            }

        function GetMentoringValues() {
            var willingParams = '';
            var count = 0;
            $('#mentoringWillings .chk').each(function() {
                willingParams += 'MentoringWillingChk[' + count + '].Id=' + $(this).val() + '&amp;';
                willingParams += 'MentoringWillingChk[' + count + '].IsChecked=' + $(this).is(':checked') + '&amp;';
                count++;
            });
            return willingParams;
        }

        function initMentoringWillings() {
            var uncheckedClass = "checkboxArea";
            var checkedClass = "checkboxAreaChecked";
            $('#chbMentor').bind("change", function() {
                var isChecked = $(this).is(':checked');
                $('#mentoringWillings .chk').each(function() {
                    if (!isChecked) {
                        $(this).prop("checked", false);
                        $(this).prev().removeClass().addClass(uncheckedClass);
                    } else {
                        $(this).prop("checked", true);
                        $(this).prev().removeClass().addClass(checkedClass);
                    }
                });
            });

            $('#mentoringWillings .chk').bind("change", function() {
                var defMentorIsChecked = $('#chbMentor').is(":checked");
                var isMainMentorChecked = false;
                $('#mentoringWillings .chk').each(function() {
                    var isChecked = $(this).is(':checked');
                    if (isChecked) {
                        isMainMentorChecked = isChecked;
                    }
                });
                if (defMentorIsChecked != isMainMentorChecked) {
                    if (!isMainMentorChecked) {
                        $('#chbMentor').prop("checked", false);
                        $('#chbMentor').prev().removeClass().addClass(uncheckedClass);
                        $('#willingFlag').val(parseInt($('#willingFlag').val()) - parseInt($('#chbMentor').val()));
                    } else {
                        $('#chbMentor').prop("checked", true);
                        $('#chbMentor').prev().removeClass().addClass(checkedClass);
                        $('#willingFlag').val(parseInt($('#willingFlag').val()) + parseInt($('#chbMentor').val()));
                    }
                }

            });
        }
