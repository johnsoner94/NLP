    $('#counselor-dropdown').on('change', function() {

            activeItems = [];
        $.each($('[data-counselor-item]'), function(i, v) {
            if (filter) {
            	var possibilities = $(this).attr('data-counselor-locations').split(',');
                $.each(possibilities, function(i, v) {
                    v = v.trim();
                    possibilities[i] = v;
                });
                if (possibilities.indexOf(filter) &gt; -1) {
                    $(this).fadeIn();
                    activeItems.push(v);
                } else {
                    $(this).fadeOut();
                }
            } else {
                activeItems.push(v);
                $(this).fadeIn();
            }
        });
        if (!activeItems.length) {
        	$('.no-counselors').fadeIn();
            $('.no-counselors .location').text(filter);
        } else {
            $('.no-counselors').fadeOut();
        }
    });
