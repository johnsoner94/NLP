        function getWIIcon(iconCode) {
            for (var i =  0; i &lt; iconCodeMap.length; i ++) {
                if (iconCode == iconCodeMap[i].iconCode) {
                    if (iconSet != '#1') {
                        return iconCodeMap[i].lwiIcon;
                    } else {
                        return iconCodeMap[i].wiIcon;
                    }
                    
                }
            }
        }
        
        function getFullWI(iconCode, size) {
            if (iconSet != '#1') {
                return 'lwi ' + getWIIcon(iconCode) + ' ' + size;
            } else {
                return 'wi ' + getWIIcon(iconCode);
            }
        }

        function getIconStyle(size){

                } else {

		function getTimeZoneOffset(hourlyForecast) {
			return new Date(hourlyForecast.validDateLocal).getTime() - new Date(hourlyForecast.validDateUtc).getTime();

		function getCurrentLocalTime(hourlyForecast) {
		    var timeZoneOffset = getTimeZoneOffset(hourlyForecast);
			var currentDate = new Date($.ajax({async: false}).getResponseHeader("Date"));
			var currentTime = currentDate.getTime();
			var offset=currentDate.getTimezoneOffset()*60000;
			var localTime = currentTime + offset + timeZoneOffset;

			return currentLocalTime;
		}
		
		function getTonightWeather(dailyForcastVOList, hourlyForcastVOList, currentLocalTime) {
		    var tonightWeather = {};
			var weekList = ["Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"];
			var todayDate = new Date(currentLocalTime.getFullYear(), currentLocalTime.getMonth(), currentLocalTime.getDate());
			var todayDayOfWeek = weekList[todayDate.getDay()];
			var tonightDate = new Date(currentLocalTime.getFullYear(), currentLocalTime.getMonth(), currentLocalTime.getDate(), 19);
			for (var i = 0; i &lt; hourlyForcastVOList.length; i++) {
				var hourlyWeather = hourlyForcastVOList[i];
				var validDateLocalDate = new Date(hourlyWeather.validDateLocal)
				if (validDateLocalDate &gt; currentLocalTime) {
					if (validDateLocalDate &gt;= tonightDate) {
						tonightWeather.hourlyWeatherTonight = hourlyWeather;
						break;
					}
				}
				
			}
			
			for (var i = 0; i &lt; dailyForcastVOList.length; i++) {
				var dailyWeather = dailyForcastVOList[i];
				if (todayDayOfWeek == dailyWeather.dayOfWk) {
					tonightWeather.todayWeather = dailyWeather;
					break;
				}
				
			}
			
			return tonightWeather;
		}
