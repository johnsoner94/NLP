    var cookiePolicyBannerComponent = (function () {
        var cookiePolicyBannerLink = $('#cookiePolicyBannerLink');
        var cookiePolicyBannerClose = $('#cookiePolicyBannerClose');
        var cookiePolicyBanner = $('#cookiePolicyBanner');
        var keyCookiePolicy = 'shouldShowCookiePolicyBanner';

        function init() {
            cookiePolicyBannerLink.click(approveCookiePolicy);
            cookiePolicyBannerClose.click(approveCookiePolicy);

            checkApproval();
        }

        function checkApproval() {
            if (shouldShowCookiePolicyBanner()) {
                cookiePolicyBanner.show();

                return;
            }

            cookiePolicyBanner.slideUp('slow');
        }

        function shouldShowCookiePolicyBanner() {
            if (localStorage &amp;&amp; localStorage[keyCookiePolicy] === 'false') {
                return false;
            }

            localStorage[keyCookiePolicy] = true;

            return true;
        }

        function approveCookiePolicy() {
            localStorage.setItem(keyCookiePolicy, false);
            checkApproval();
        }

        return {
            init: init,
            checkApproval: checkApproval
        };
    })();

    $(document).ready(function () {
